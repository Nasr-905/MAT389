\begin{example}

    \begin{align*}
        \frac{z^2}{\sin z}                               \\
        z_0 = n\pi                                       \\
        \int_{|z -z_0| = 1} \frac{z^2}{\sin z} dz \neq 0 \\
        \int_{|z| = 1} \frac{z^2}{\sin z} dz = 0
    \end{align*}
    because
    \begin{align*}
        \lim_{z \to 0} \frac{\sin z}{z} = 1
    \end{align*}
\end{example}
